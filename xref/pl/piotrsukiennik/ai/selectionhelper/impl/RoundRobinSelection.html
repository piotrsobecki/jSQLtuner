12   * @author Piotr Sukiennik

14   */
15  public class RoundRobinSelection<T extends Selectable> extends CollectionSelection<T> implements UpdateableSelectionHelper<T> {
16  
17      private LinkedHashSet<T> selectableSet = new LinkedHashSet<T>();
18  
19      private Iterator<T> selectableIterator = selectableSet.iterator();
20  
21      @Override
22      protected Set<T> getCollection() {
23          return selectableSet;
24      }
25  
26      public synchronized boolean update( T value ) {
27          if ( super.update( value ) ) {
28              reset();
29              return true;
30          }
31          return false;
32      }
33  
34      public void reset() {
35          selectableIterator = selectableSet.iterator();
36      }
37  
38      @Override
39      public synchronized T select() {
40          if ( !selectableIterator.hasNext() ) {
41              reset();
42          }
43          if ( selectableIterator.hasNext() ) {
44              return selectableIterator.next();
45          }
46          return null;
47      }
48  
49      @Override
50      public synchronized void submit( T selectable ) {
51          selectableSet.add( selectable );
52          reset();
53      }
54  }
