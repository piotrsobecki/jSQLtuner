15  @Service
16  public class HashesHolderImpl<P, K> implements HashesHolder<P, K> {
17      private HolderEnterableFactory holderEnterableFactory;
18  
19      private HolderEnterable<P, K> rootHolder;
20  
21      public HashesHolderImpl() {
22          this( new HolderEnterableFactoryImpl( LinkedHashSet.class ) );
23      }
24  
25      public HashesHolderImpl( HolderEnterableFactory holderEnterableFactory ) {
26          this.rootHolder = holderEnterableFactory.buildEnterable();
27          this.holderEnterableFactory = holderEnterableFactory;
28      }
29  
30      @Override
31      public void put( P[] path, K hash ) {
32          Holder holder = rootHolder;
33          for ( int i = 0; i < path.length; i++ ) {
34              if ( holder instanceof HolderEnterable ) {
35                  Holder holderNew = ( (HolderEnterable<P, K>) holder ).enter( path[i] );
36                  if ( holderNew == null ) {
37                      holderNew = holderEnterableFactory.buildEnterable();
38                      ( (HolderEnterable) holder ).put( path[i], holderNew );
39                  }
40                  holder = holderNew;
41              }
42              holder.add( hash );
43          }
44      }
45  
46      @Override
47      public Collection<K> get( P[] path ) {
48          Holder holder = rootHolder;
49          for ( P str : path ) {
50              if ( holder instanceof HolderEnterable ) {
51                  holder = ( (HolderEnterable<P, K>) holder ).enter( str );
52              }
53          }
54          if ( holder == null ) {
55              return Collections.EMPTY_LIST;
56          }
57          return holder.get();
58      }
59  
60      @Override
61      public Collection<K> getRemove( P[] path ) {
62          Holder holder = rootHolder;
63          for ( P str : path ) {
64              if ( holder != null && holder instanceof HolderEnterable ) {
65                  holder = ( (HolderEnterable<P, K>) holder ).enter( str );
66              }
67          }
68          if ( holder != null ) {
69              Collection<K> objects = holder.get();
70              rootHolder.remove( objects );
71              return objects;
72          }
73          return Collections.EMPTY_LIST;
74      }

77  }
