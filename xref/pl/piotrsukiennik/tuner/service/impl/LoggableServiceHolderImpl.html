24  @Service("pl.piotrsukiennik.tuner.LoggableService.impl")
25  public class LoggableServiceHolderImpl extends LoggableServiceHolder implements LoggableService {
26  
27      private static final Log LOG = LogFactory.getLog( LoggableServiceHolder.class );
28  
29      private static final String EXCEPTION_MESSAGE_FORMAT = "Query (%s) caused exception.";
30  
31      public LogDao getLogDao() {
32          return Dao.getLog();
33      }
34  
35      public LoggableServiceHolderImpl() {
36          LoggableServiceHolder.setLogService( this );
37      }
38  
39      @Override
40      public void log( DataRetrieval dataRetrieval ) {
41          if ( LOG.isInfoEnabled() ) {
42              LOG.info( LoggableMessageEnum.EXECUTION.getMessage( dataRetrieval ) );
43          }
44      }
45  
46      @Override
47      public void log( Query query, LoggableMessage loggableMessage, TimeUnit timeUnit, long duration ) {
48          if ( LOG.isInfoEnabled() ) {
49              LOG.info( loggableMessage.getMessage( query, timeUnit, duration ) );
50          }
51      }
52  
53  
54      private void log( String query, Throwable exception ) {
55          if ( LOG.isWarnEnabled() ) {
56              LOG.warn( String.format( EXCEPTION_MESSAGE_FORMAT, query ), exception );
57          }
58      }
59  
60  
61      @Override
62      public void log( DataRetrievalException exception ) {
63          log( exception.getQuery(), exception );
64          getLogDao().log( exception.getQuery(), exception );
65      }
66  
67      @Override
68      public void log( QueryInterceptionNotSupportedException exception ) {
69          log( exception.getQuery(), exception );
70      }
71  
72      @Override
73      public void log( QueryParsingNotSupportedException exception ) {
74          log( exception.getQuery(), exception );
75      }
76  }
