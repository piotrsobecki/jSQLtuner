21  public class SelectBodyParser<T extends SelectQuery> extends ParsingVisitor<T> implements net.sf.jsqlparser.statement.select.SelectVisitor {
22  
23      public SelectBodyParser( JsqlParserQueryParsingContext parsingContext, T query ) {
24          super( parsingContext, query );
25      }
26  
27      @Override
28      public void visit( PlainSelect plainSelect ) {
29          FromItemParser fromItemParser = new FromItemParser( parsingContext );
30          plainSelect.getFromItem().accept( fromItemParser );
31          NewQueryUtils.addSource( query, fromItemParser.getSource() );
32          if ( plainSelect.getFromItem() != null ) {
33              plainSelect.getFromItem().accept( fromItemParser );
34          }
35  
36          List<SelectItem> selectItems = plainSelect.getSelectItems();
37          SelectItemParser selectItemParser = new SelectItemParser( parsingContext, query );
38          for ( SelectItem selectItem : selectItems ) {
39              selectItem.accept( selectItemParser );
40          }
41  
42          ExpresionParser expresionParser = new ExpresionParser( parsingContext );
43  
44          if ( plainSelect.getHaving() != null ) {
45              plainSelect.getHaving().accept( expresionParser );
46              query.setHavingExpression( (OperatorExpression) expresionParser.getExpression() );
47          }
48  
49          if ( plainSelect.getWhere() != null ) {
50              plainSelect.getWhere().accept( expresionParser );
51              query.setWhereExpression( (OperatorExpression) expresionParser.getExpression() );
52          }
53  
54          if ( plainSelect.getJoins() != null ) {
55              List<Join> joins = plainSelect.getJoins();
56              for ( Join join : joins ) {
57                  JoinFragment joinFragment = parsingContext.getJoin( join );
58                  NewQueryUtils.addJoin( query, joinFragment );
59              }
60          }
61  
62          if ( plainSelect.getOrderByElements() != null ) {
63              List<OrderByElement> orderByElements = plainSelect.getOrderByElements();
64              for ( int i = 0; i < orderByElements.size(); i++ ) {
65                  OrderByElement orderByElement = orderByElements.get( i );
66                  OrderByFragment orderByFragment = parsingContext.getOrderByFragment( orderByElement );
67                  NewQueryUtils.addOrderByFragment( query, orderByFragment );
68              }
69  
70          }
71  
72          if ( plainSelect.getGroupByColumnReferences() != null ) {
73              List<Expression> groupByColumns = plainSelect.getGroupByColumnReferences();
74              for ( Expression column : groupByColumns ) {
75                  pl.piotrsukiennik.tuner.model.expression.Expression fragment = null;
76                  if ( column instanceof Column ) {
77                      fragment = parsingContext.getColumnProjection( (Column) column );
78                  }
79                  else {
80                      ExpresionParser groupByColumnExpresionParser = new ExpresionParser( parsingContext );
81                      column.accept( expresionParser );
82                      fragment = groupByColumnExpresionParser.getExpression();
83                  }
84                  NewQueryUtils.addGroupByFragment( query, fragment );
85              }
86          }
87  
88          if ( plainSelect.getInto() != null ) {
89              query.setIntoTable( parsingContext.getTable( plainSelect.getInto() ) );
90          }
91  
92          if ( plainSelect.getLimit() != null ) {
93              Limit limit = plainSelect.getLimit();
94              query.setLimitFrom( limit.getOffset() );
95              query.setLimitTo( limit.getOffset() + limit.getRowCount() );
96          }
97  
98          init( plainSelect );
99  
100     }
