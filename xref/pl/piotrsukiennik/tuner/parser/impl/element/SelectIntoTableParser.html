16  public class SelectIntoTableParser extends Visitor implements IntoTableVisitor {
17  
18      private SelectQuery sourceQuery;
19  
20  
21      public SelectIntoTableParser( JsqlParserQueryParsingContext parsingContext, SelectQuery sourceQuery ) {
22          super( parsingContext );
23          this.sourceQuery = sourceQuery;
24      }
25  
26      @Override
27      public void visit( Table tableName ) {
28          TableSource tableSource = parsingContext.getTableSource( tableName );
29          NewQueryUtils.addSource( sourceQuery, tableSource );
30      }

32  }
