28  @Service
29  class ParserImpl implements Parser {
30  
31      //private static final Log LOG = LogFactory.getLog( Parser.class );
32  
33      @Autowired
34      private QueryParsingContextBuilder<JsqlParserQueryParsingContext> queryParsingContextBuilder;
35  
36      @Autowired
37      private LoggableService logger;
38  
39  
40      @Override
41      public <T extends Query> T parse( String database, String schema, String query ) throws QueryParsingNotSupportedException {
42          T parsedQuery;
43          TimedCallable<T> timedCallable = getParserCallable( database, schema, query );
44          try {
45              parsedQuery = timedCallable.call();
46              logger.log( parsedQuery, LoggableMessageEnum.PARSING, TimeUnit.NANOSECONDS, timedCallable.getTime( TimeUnit.NANOSECONDS ) );
47              return parsedQuery;
48          }
49          catch ( Exception e ) {
50              throw new QueryParsingNotSupportedException( e, query );
51          }
52      }
53  
54  
55      protected <T extends Query> TimedCallable<T> getParserCallable( final String database, final String schema, final String query ) throws QueryParsingNotSupportedException {
56          return new TimedCallableImpl<T>( new Callable<T>() {
57              @Override
58              public T call() throws Exception {
59                  Statement statement = parse( query );
60                  return parse( database, schema, query, statement );
61              }
62          } );
63      }
64  
65      protected Statement parse( String query ) throws JSQLParserException {
66          CCJSqlParserManager pm = new CCJSqlParserManager();
67          return pm.parse( new StringReader( query ) );
68      }
69  
70      protected <T extends Query> T parse( String database, String schema, String query, Statement statement ) throws QueryParsingNotSupportedException {
71          JsqlParserQueryParsingContext parsingContext = queryParsingContextBuilder.getQueryContext( database, schema );
72          StatementParserVisitor<T> statementParserVisitor = new StatementParserVisitor<T>( parsingContext, statement );
73          T mappedQuery = (T) statementParserVisitor.getQuery();
74          if ( mappedQuery == null ) {
75              throw new QueryParsingNotSupportedException( query );
76          }
77          return mappedQuery;
78      }
79  
80  }
