4   import org.springframework.stereotype.Repository;
5   import pl.piotrsukiennik.tuner.model.log.WriteQueryExecution;

14  @Repository
15  class QueryDaoImpl extends CrudDaoImpl implements QueryDao {
16  
17      public <T extends Query> T getQuery( String hash ) {
18          return (T) s().createCriteria( Query.class ).add( Restrictions.eq( "hash", hash ) ).setMaxResults( 1 ).uniqueResult();
19      }
20  
21      public <T extends Query> T getQuery( Class<? extends Query> clazz, String hash ) {
22          return (T) s().createCriteria( clazz ).add( Restrictions.eq( "hash", hash ) ).setMaxResults( 1 ).uniqueResult();
23      }
24  
25      @Override
26      public <T extends Query> T createQuery( T query ) {
27          if ( query.getId() == 0 ) {
28              T queryPersisted = getQuery( query.getClass(), query.getHash() );
29              if ( queryPersisted == null ) {
30                  return create( query );
31              }
32              if ( queryPersisted.getId() != 0 ) {
33                  return queryPersisted;
34              }
35              throw new RuntimeException( "Not persisted Query" );
36          }
37          else {
38              return query;
39          }
40      }
41  
42      @Override
43      public void createQueryExecution( WriteQueryExecution writeQueryExecution ) {
44          create( writeQueryExecution );
45      }
46  }
