13  abstract class CrudDaoImpl extends AbstractDaoImpl {
14  
15      public <T extends ValueEntity> T create( T t ) {
16          Session session = s();
17          Serializable serializable = session.save( t );
18          t.setId( (Long) serializable );
19          session.flush();
20          return t;
21      }
22  
23      public <T extends ValueEntity> T read( Long id ) {
24          throw new RuntimeException( "Read(id) NOT SUPPORTED" );
25      }
26  
27      public <T extends ValueEntity> T update( T t ) {
28          Session session = s();
29          T tmerged = (T) session.merge( t );
30          session.flush();
31          session.refresh( tmerged );
32          return tmerged;
33      }
34  
35      public <T extends ValueEntity> void delete( T t ) {
36          Session session = s();
37          session.delete( t );
38          session.flush();
39      }
40  }
